// Referentially Transparent Function: add
// Takes two numbers as inputs and returns their sum.
function add(a, b) {
  return a + b;
}

// The function call can be replaced with its result (5):
const result1 = add(2, 3); // result1 = 5

// The function call can be replaced with its result (5):
const result2 = 5; // result2 = 5
